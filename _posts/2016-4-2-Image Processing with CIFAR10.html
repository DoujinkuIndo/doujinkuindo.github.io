def steps(n,stride,window):
    return int((n - window )/stride + 1) 

def steps_inverse(m,stride,window):
    return ((m-1)*stride + window)

def zeroPad(image, p = None,even = True):
    if even:
        return np.lib.pad(image,((0,0),(0,0),(p,p),(p,p)),'constant', constant_values=0)
    else:
        return np.lib.pad(image,((0,0),(0,0),(p,0),(p,0)),'constant', constant_values=0)

class Kernels(): #has a shape, weights, biases and a stride
    def __init__(self,shape,stride = 2):
        self.shape = shape
        self.weights = np.random.standard_normal(shape)
        self.biases = np.random.standard_normal(self.shape[0]) #???
        
        self.stride = stride
        self.k2,self.k1,_,self.w = self.shape
        
    def fdot(self,x): 
        return  np.tensordot(x,self.weights,[[1,2,3],[1,2,3]]) + self.biases 
    
    def convolve(self,x):
        N,_,_,n = x.shape
        self.m = steps(n,self.stride,self.w)
        Results = np.zeros((N,self.k2,self.m,self.m))
        self.Inputs = np.zeros((N,self.m,self.m,self.k1,self.w,self.w))
        for i in range(self.m):
            I = i*self.stride
            for j in range(self.m):
                J = j*self.stride
                self.Inputs[:,i,j,:,:,:] = x[:,:,I:I+self.w,J:J+self.w] #doesnt seem very space efficient?
                Results[:,:,i,j] = self.fdot(x[:,:,I:I+self.w,J:J+self.w])
        self.Inputs = np.sum(self.Inputs, axis = (0,1,2))
        return Results
    
    def deconvolve(self,x): #given a delta for each kernel
        k2,k1,_,w = self.weights.shape
        n = steps_inverse(self.m,self.stride,w)
        Results = np.zeros((k1,n,n))
        
        for i in range(self.m):
            I = i*self.stride
            for j in range(self.m):
                J = j*self.stride
                #tile the delta onto the previous image space
                #through each kernels weights
                Results[:,I:I+w,J:J+w] += np.tensordot(self.weights,x,[[0],[0]]).squeeze() #biases??
        return Results

def edgedetector(n):
    im = batch1['data'][n].reshape( (1,32,32,3), order='F')
    
    # an edge detector
    shape = (1,3,3,3)
    K = Kernels(shape,stride = 1)
    K.weights = np.zeros(shape)
    K.weights[:,:,1,0] = 1
    K.weights[:,:,1,1] = -1
    #convolve the edge detector with the image
    image1 = K.convolve(np.transpose(im,(0,3,1,2))) 
    
    plotCIFAR(im.reshape(32,32,3))
    plotCIFAR(image1.reshape(30,30),colourmap = cm.Greys_r) 
    print('s')
