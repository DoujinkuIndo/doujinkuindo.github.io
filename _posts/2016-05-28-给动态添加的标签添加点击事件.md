###动态添加即通过js(ajax)添加的html代码，要给它添加事件，比如点击事件，需要在回调函数中添加
```javascritp
    //ajax返回台站信息（通过传入学科名称异步返回台站信息）
    function returnStations(sub){
        $.ajax({
            type: "POST",
            url: "{:U('Home/Index/ajax_return_stations')}",
            datatype: "json",
            data: {"sub":sub},
            success: function (data) {
                var json = eval('(' + data + ')');
                var htmlStr = "<ul class='list'>";
                for (var i = 0; i < json.length; i++) {
                    var VALUE=json[i].STATIONID.trim()+"_"+json[i].STATIONNAME.trim();
                    htmlStr += "<li class='tableTemp' value='"+VALUE+"' station='"+VALUE+"'>"+VALUE +"</li>";
                }
                htmlStr +="</ul>"
                $("#select-left-2-body").html(htmlStr);
                $(".list li").click(function(){//在这里为动态添加的li添加点击事件
                    $(this).addClass("cur").siblings().removeClass("cur");
                    $("#select-left-3-body").html("");
                    returnPoints($(this).html());
                })
            }
        });
    }
```
